191 // Is IP address NULL/invalid/unspecified?
192 // There are two variants being used as invalid/unspecified address:
193 // 00000000 00000000 00000000 ffffffff (i.e. "0.0.0.0", zero IPv4)
194 // 00000000 00000000 00000000 00000000 (i.e. "::", zero IPv6)
195 // This function returns 1 if given address matches any of the ones above.
196 // Otherwise returns 0.
197 INLINE int ip_is_null(const ip_addr_t *addr)
198 {
199  if (addr->ui64[0] == 0) {
200  if (addr->ui64[1] == 0 || (addr->ui32[2] == 0 && addr->ui32[3] == 0xffffffff)) {
201  return 1;
202  }
203  }
204  return 0;
205 }

208 // Convert IP address in a string into ip_addr_t. Return 1 on success,
209 // 0 on error (i.e. string doesn't contain a valid IP address).
210 INLINE int ip_from_str(const char *str, ip_addr_t *addr)
211 {
212  char tmp[16];
213  if (strchr(str, ':') == NULL) { // IPv4
214  if (inet_pton(AF_INET, str, (void *) tmp) != 1) {
215  return 0; // err
216  }
217  *addr = ip_from_4_bytes_be(tmp);
218  return 1;
219  } else { // IPv6
220  if (inet_pton(AF_INET6, str, (void *) tmp) != 1) {
221  return 0; // err
222  }
223  *addr = ip_from_16_bytes_be(tmp);
224  return 1;
225  }
226 }
